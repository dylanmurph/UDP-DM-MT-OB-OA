Setting up pi for nfc:

sudo raspi-config      # Interface Options → I2C → Enable → Finish → Reboot

sudo apt update
sudo apt install -y i2c-tools python3-pip
pip3 install --break-system-packages adafruit-circuitpython-pn532

python3 -m venv venv
pip install adafruit-circuitpython-pn532 RPi.GPIO flask
python nfc_test.py
ctn+z
deactivate
